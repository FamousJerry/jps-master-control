    <script>

        const users = {

            admin: { name: "John 'JPS' Smith", role: "Administrator" },

            Alice: { name: "Alice", role: "Team Member" },

            Bob: { name: "Bob", role: "Team Member" }

        };

        const jobs = [

            { id: 1, date: '2025-08-11', title: 'Client Photoshoot', client: 'Quantum Solutions', assignedTo: ['Alice', 'Bob'], status: 'completed' },

            { id: 2, date: '2025-08-11', title: 'Equipment Prep', client: 'Internal Task', assignedTo: ['Charlie'], status: 'completed' },

            { id: 3, date: '2025-08-12', title: 'Video Edit Session 1', client: 'Stellar Designs', assignedTo: ['Diana'], status: 'inprogress' },

            { id: 4, date: '2025-08-12', title: 'On-site filming', client: 'Apex Industries', assignedTo: ['Alice', 'Evan'], status: 'planned' },

            { id: 5, date: '2025-08-13', title: 'Pre-production Call', client: 'Nebula Corp', assignedTo: ['John'], status: 'planned' },

            { id: 6, date: '2025-08-14', title: 'Drone Shoot (Weather Hold)', client: 'Fusion Works', assignedTo: ['Bob', 'Charlie'], status: 'problem' },

            { id: 7, date: '2025-08-15', title: 'Final Video Delivery', client: 'Quantum Solutions', assignedTo: ['Diana'], status: 'planned' },

            { id: 8, date: '2025-08-20', title: 'New Project Kickoff', client: 'Evergreen Holdings', assignedTo: ['John', 'Alice'], status: 'planned' },

        ];

        let currentUser = 'admin';

        let currentView = 'week';

        let currentDate = new Date('2025-08-13T12:00:00Z');

        const calendarContainer = document.getElementById('calendar-container');

        const weekViewBtn = document.getElementById('weekViewBtn');

        const monthViewBtn = document.getElementById('monthViewBtn');

        const userRoleSwitcher = document.getElementById('userRoleSwitcher');

        const currentDateDisplay = document.getElementById('currentDateDisplay');

        const jobModal = document.getElementById('jobModal');

        const jobModalContent = document.getElementById('jobModalContent');

        function renderCalendar() {

            calendarContainer.innerHTML = '';

            if (currentView === 'week') {

                renderWeeklyView();

            } else {

                renderMonthlyView();

            }

        }

        function renderWeeklyView() {

            currentDateDisplay.textContent = `Week of ${currentDate.toLocaleDateString('en-US', { month: 'long', day: 'numeric' })}`;

            const weekGrid = document.createElement('div');

            weekGrid.className = 'calendar-grid-week';

            const startOfWeek = new Date(currentDate);

            startOfWeek.setDate(startOfWeek.getDate() - startOfWeek.getDay() + 1);

            for (let i = 0; i < 7; i++) {

                const day = new Date(startOfWeek);

                day.setDate(day.getDate() + i);

                const dayColumn = document.createElement('div');

                dayColumn.className = 'day-column';

                dayColumn.innerHTML = `<h4 class="font-bold text-center mb-4">${day.toLocaleDateString('en-US', { weekday: 'long', month: 'short', day: 'numeric' })}</h4>`;

                const jobsForDay = jobs.filter(job => new Date(job.date).toDateString() === day.toDateString());

                jobsForDay.forEach(job => {

                    dayColumn.appendChild(createJobCard(job));

                });

                weekGrid.appendChild(dayColumn);

            }

            calendarContainer.appendChild(weekGrid);

        }

        function renderMonthlyView() {

            currentDateDisplay.textContent = currentDate.toLocaleDateString('en-US', { month: 'long', year: 'numeric' });

            const monthGrid = document.createElement('div');

            monthGrid.className = 'calendar-grid-month';

            const year = currentDate.getFullYear();

            const month = currentDate.getMonth();

            const firstDay = new Date(year, month, 1);

            const lastDay = new Date(year, month + 1, 0);

            const startOffset = (firstDay.getDay() + 6) % 7;

            for (let i = 0; i < startOffset; i++) {

                const dayCell = document.createElement('div');

                dayCell.className = 'month-day not-current-month';

                monthGrid.appendChild(dayCell);

            }

            for (let i = 1; i <= lastDay.getDate(); i++) {

                const day = new Date(year, month, i);

                const dayCell = document.createElement('div');

                dayCell.className = 'month-day';

                dayCell.innerHTML = `<div class="day-number">${i}</div>`;

                const jobsForDay = jobs.filter(job => new Date(job.date).toDateString() === day.toDateString());

                const jobsContainer = document.createElement('div');

                jobsForDay.forEach(job => {

                    const dot = document.createElement('span');

                    dot.className = 'month-job-dot';

                    dot.style.backgroundColor = getStatusColor(job.status);

                    if (job.assignedTo.includes(currentUser)) {

                        dot.classList.add('ring-2', 'ring-green-400');

                    }

                    jobsContainer.appendChild(dot);

                });

                dayCell.appendChild(jobsContainer);

                monthGrid.appendChild(dayCell);

            }

            calendarContainer.appendChild(monthGrid);

        }

        function createJobCard(job) {

            const card = document.createElement('div');

            card.className = `job-card status-${job.status}`;

            card.dataset.jobId = job.id;

            if (currentUser !== 'admin' && job.assignedTo.includes(currentUser)) {

                card.classList.add('my-job');

            }

            card.innerHTML = `

                <p class="font-bold">${job.title}</p>

                <p class="text-sm text-gray-300">${job.client}</p>

                <p class="text-xs text-gray-400 mt-2">Team: ${job.assignedTo.join(', ')}</p>

            `;

            card.addEventListener('click', () => openJobModal(job.id));

            return card;

        }

        function getStatusColor(status) {

            switch(status) {

                case 'completed': return '#22c55e';

                case 'inprogress': return '#eab308';

                case 'problem': return '#ef4444';

                default: return '#4f46e5';

            }

        }

        function openJobModal(jobId) {

            const job = jobs.find(j => j.id === jobId);

            if (!job) return;

            let actionsHtml = '';

            if (currentUser === 'admin') {

                actionsHtml = `

                    <button class="bg-blue-600 hover:bg-blue-700 text-white font-bold py-2 px-4 rounded">Edit Job</button>

                    <button class="bg-red-600 hover:bg-red-700 text-white font-bold py-2 px-4 rounded">Delete Job</button>

                `;

            } else if (job.assignedTo.includes(currentUser)) {

                actionsHtml = `<button class="bg-yellow-600 hover:bg-yellow-700 text-white font-bold py-2 px-4 rounded">Request Change</button>`;

            }

            jobModalContent.innerHTML = `

                <div class="p-6">

                    <div class="flex justify-between items-start">

                        <div>

                            <h3 class="text-2xl font-bold">${job.title}</h3>

                            <p class="text-lg text-gray-400">${job.client}</p>

                        </div>

                        <button id="closeModalBtn" class="text-2xl">&times;</button>

                    </div>

                    <div class="my-4 border-t border-gray-700"></div>

                    <div class="space-y-2">

                        <p><strong>Date:</strong> ${new Date(job.date).toLocaleDateString('en-US', { dateStyle: 'full' })}</p>

                        <p><strong>Status:</strong> <span class="font-semibold capitalize" style="color: ${getStatusColor(job.status)}">${job.status}</span></p>

                        <p><strong>Assigned Team:</strong> ${job.assignedTo.join(', ')}</p>

                    </div>

                    <div class="my-4 border-t border-gray-700"></div>

                    <div class="flex justify-end space-x-4">

                        ${actionsHtml}

                    </div>

                </div>

            `;

            jobModal.classList.remove('hidden');

            jobModal.classList.add('flex');

            document.getElementById('closeModalBtn').addEventListener('click', () => {

                jobModal.classList.add('hidden');

                jobModal.classList.remove('flex');

            });

        }

        weekViewBtn.addEventListener('click', () => {

            currentView = 'week';

            weekViewBtn.classList.add('active');

            monthViewBtn.classList.remove('active');

            renderCalendar();

        });

        monthViewBtn.addEventListener('click', () => {

            currentView = 'month';

            monthViewBtn.classList.add('active');

            weekViewBtn.classList.remove('active');

            renderCalendar();

        });

        userRoleSwitcher.addEventListener('change', (e) => {

            currentUser = e.target.value;

            const user = users[currentUser];

            document.getElementById('userName').textContent = user.name;

            document.getElementById('userRole').textContent = user.role;

            renderCalendar();

        });

        document.addEventListener('DOMContentLoaded', () => {

            renderCalendar();

        });

    </script>
